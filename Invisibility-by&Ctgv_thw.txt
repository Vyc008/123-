local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local player = Players.LocalPlayer

-- ==================== CLEANUP HỆ THỐNG CŨ ====================
if _G.NuclearBobo5_Cleanup then
    pcall(_G.NuclearBobo5_Cleanup)
end

local character, humanoid, rootPart
local invisible = false
local parts = {}
local connections = {}
local uiVisible = true
local gui, mainFrame, toggleOpenBtn

_G.NuclearBobo5_Cleanup = function()
    for _, conn in pairs(connections) do
        pcall(function() conn:Disconnect() end)
    end
    if gui then
        pcall(function() gui:Destroy() end)
    end
end

-- CẤU HÌNH PHÍM TẮT
local Config = {
    ToggleKey = Enum.KeyCode.X,
    UIToggleKey = Enum.KeyCode.M,
    TeleportHeight = -200000,
    Transparency = 0.7,
}

-- ==========================================
-- HÀM HỖ TRỢ LÀM ĐẸP (MODERN UI)
-- ==========================================
local function applyStyle(instance, radius)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius or 6)
    corner.Parent = instance
    
    if instance:IsA("GuiButton") or instance:IsA("Frame") then
        local stroke = Instance.new("UIStroke")
        stroke.Color = Color3.fromRGB(60, 60, 65)
        stroke.Thickness = 1.2
        stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
        stroke.Parent = instance
    end
end

-- ==========================================
-- LOGIC TÀNG HÌNH
-- ==========================================
local function setupCharacter()
    character = player.Character or player.CharacterAdded:Wait()
    humanoid = character:WaitForChild("Humanoid")
    rootPart = character:WaitForChild("HumanoidRootPart")
    parts = {}
    
    for _, obj in pairs(character:GetDescendants()) do
        if obj:IsA("BasePart") then
            table.insert(parts, obj)
        end
    end
    
    if invisible then
        for _, part in pairs(parts) do
            part.Transparency = Config.Transparency
        end
    end
end

local function updateUI()
    if mainFrame then
        local toggleBtn = mainFrame:FindFirstChild("ToggleBtn")
        local statusLabel = mainFrame:FindFirstChild("StatusLabel")
        
        if toggleBtn then
            if invisible then
                toggleBtn.Text = "TÀNG HÌNH: BẬT"
                toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 150, 50)
            else
                toggleBtn.Text = "TÀNG HÌNH: TẮT"
                toggleBtn.BackgroundColor3 = Color3.fromRGB(80, 80, 85)
            end
        end
        
        if statusLabel then
            statusLabel.Text = invisible and "Trạng thái: 🟢" or "Trạng thái: 🔴"
        end
    end
end

local function toggleInvisibility()
    invisible = not invisible
    for _, part in pairs(parts) do
        if part:IsA("BasePart") then
            part.Transparency = invisible and Config.Transparency or 0
        end
    end
    updateUI()
end

local function toggleUI()
    uiVisible = not uiVisible
    if mainFrame then
        mainFrame.Visible = uiVisible
    end
    if toggleOpenBtn then
        toggleOpenBtn.Text = uiVisible and "👉👌" or "🥵"
    end
end

-- ==========================================
-- TẠO GIAO DIỆN SIÊU NHỎ GỌN (ULTRA COMPACT)
-- ==========================================
local function createMainUI()
    if gui then gui:Destroy() end
    
    gui = Instance.new("ScreenGui")
    gui.Name = "BatmanGuiUI"
    gui.ResetOnSpawn = false
    gui.Parent = player.PlayerGui
    
    -- MAIN FRAME (Thu nhỏ đáng kể kích thước)
    mainFrame = Instance.new("Frame")
    mainFrame.Size = UDim2.new(0, 140, 0, 95)
    mainFrame.Position = UDim2.new(0.5, -70, 0.4, 0)
    mainFrame.BackgroundColor3 = Color3.fromRGB(30, 30, 35)
    mainFrame.Active = true
    mainFrame.Draggable = false 
    mainFrame.Parent = gui
    applyStyle(mainFrame, 10)
    
    -- TITLE (Đổi tên thành Batman Gui <Ctgv-thw>)
    local title = Instance.new("TextLabel")
    title.Size = UDim2.new(1, 0, 0, 28)
    title.Text = "Batman Gui <Ctgv-thw>"
    title.TextColor3 = Color3.fromRGB(255, 255, 255)
    title.Font = Enum.Font.GothamBold
    title.TextSize = 11
    title.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
    title.Active = true
    title.Parent = mainFrame
    applyStyle(title, 10)
    
    -- STATUS HINT (Rút ngắn văn bản để vừa khung nhỏ)
    local statusLabel = Instance.new("TextLabel")
    statusLabel.Name = "StatusLabel"
    statusLabel.Size = UDim2.new(1, 0, 0, 18)
    statusLabel.Position = UDim2.new(0, 0, 0, 32)
    statusLabel.BackgroundTransparency = 1
    statusLabel.Text = "Trạng thái: 🔴"
    statusLabel.TextColor3 = Color3.fromRGB(180, 180, 180)
    statusLabel.Font = Enum.Font.GothamMedium
    statusLabel.TextSize = 11
    statusLabel.Parent = mainFrame
    
    -- HOTKEY HINT (Làm mờ nhỏ gọn bên dưới)
    local hotkeyLabel = Instance.new("TextLabel")
    hotkeyLabel.Size = UDim2.new(1, 0, 0, 12)
    hotkeyLabel.Position = UDim2.new(0, 0, 0, 50)
    hotkeyLabel.BackgroundTransparency = 1
    hotkeyLabel.Text = "Phím tắt: X | Ẩn: M"
    hotkeyLabel.TextColor3 = Color3.fromRGB(100, 100, 100)
    hotkeyLabel.Font = Enum.Font.Gotham
    hotkeyLabel.TextSize = 9
    hotkeyLabel.Parent = mainFrame

    -- TOGGLE BUTTON
    local toggleBtn = Instance.new("TextButton")
    toggleBtn.Name = "ToggleBtn"
    toggleBtn.Size = UDim2.new(1, -20, 0, 24)
    toggleBtn.Position = UDim2.new(0, 10, 0, 64)
    toggleBtn.Text = "TÀNG HÌNH: TẮT"
    toggleBtn.BackgroundColor3 = Color3.fromRGB(80, 80, 85)
    toggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
    toggleBtn.Font = Enum.Font.GothamBold
    toggleBtn.TextSize = 10
    toggleBtn.Parent = mainFrame
    applyStyle(toggleBtn, 6)
    
    toggleBtn.MouseButton1Click:Connect(toggleInvisibility)
    
    -- NÚT ẨN/HIỆN GIAO DIỆN CHÍNH
    toggleOpenBtn = Instance.new("TextButton")
    toggleOpenBtn.Size = UDim2.new(0, 35, 0, 35)
    toggleOpenBtn.Position = UDim2.new(0, 10, 0, 60)
    toggleOpenBtn.Text = "👉👌"
    toggleOpenBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
    toggleOpenBtn.Font = Enum.Font.GothamBold
    toggleOpenBtn.TextSize = 16
    toggleOpenBtn.BackgroundColor3 = Color3.fromRGB(30, 30, 35)
    toggleOpenBtn.Parent = gui
    applyStyle(toggleOpenBtn, 8)
    
    toggleOpenBtn.MouseButton1Click:Connect(toggleUI)
    
    -- ==========================================
    -- HỆ THỐNG KÉO MENU (DRAG THEO TITLE)
    -- ==========================================
    local draggingMenu = false
    local dragInput, dragStart, startPos

    title.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            draggingMenu = true
            dragStart = input.Position
            startPos = mainFrame.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    draggingMenu = false
                end
            end)
        end
    end)

    title.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and draggingMenu then
            local delta = input.Position - dragStart
            mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end

-- ==========================================
-- TELEPORT LOOP (LÕI TÀNG HÌNH)
-- ==========================================
local function startTeleportLoop()
    connections.Heartbeat = RunService.Heartbeat:Connect(function()
        if invisible and character and rootPart and humanoid then
            local cf = rootPart.CFrame
            local camOffset = humanoid.CameraOffset
            local hidden = cf * CFrame.new(0, Config.TeleportHeight, 0)
            
            rootPart.CFrame = hidden
            humanoid.CameraOffset = hidden:ToObjectSpace(CFrame.new(cf.Position)).Position
            
            RunService.RenderStepped:Wait()
            
            rootPart.CFrame = cf
            humanoid.CameraOffset = camOffset
        end
    end)
end

-- ==========================================
-- KHỞI CHẠY VÀ KẾT NỐI SỰ KIỆN
-- ==========================================
setupCharacter()
createMainUI()
startTeleportLoop()

connections.KeyToggle = UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Config.ToggleKey then
        toggleInvisibility()
    elseif input.KeyCode == Config.UIToggleKey then
        toggleUI()
    end
end)

connections.CharacterAdded = player.CharacterAdded:Connect(function()
    invisible = false
    task.wait(0.5) 
    setupCharacter()
    updateUI()
end)

connections.PlayerLeaving = player:GetPropertyChangedSignal("Parent"):Connect(function()
    if not player.Parent then
        if _G.NuclearBobo5_Cleanup then
            pcall(_G.NuclearBobo5_Cleanup)
        end
    end
end)
