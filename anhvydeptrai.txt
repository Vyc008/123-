local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local player = Players.LocalPlayer

-- ==================== SMOOTH RAINBOW FUNCTION ====================
local function createSmoothRainbow(stroke)
    local colors = {
        Color3.new(1, 0, 0),     -- Red
        Color3.new(1, 0.5, 0),   -- Orange
        Color3.new(1, 1, 0),     -- Yellow
        Color3.new(0, 1, 0),     -- Green
        Color3.new(0, 0.5, 1),   -- Light Blue
        Color3.new(0.5, 0, 1),   -- Purple
        Color3.new(1, 0, 1)      -- Pink
    }
    
    local currentIndex = 2
    local nextIndex = 3
    local progress = 0
    
    while stroke and stroke.Parent do
        local currentColor = colors[currentIndex]
        local nextColor = colors[nextIndex]
        
        -- Interpolate between colors
        local r = currentColor.R + (nextColor.R - currentColor.R) * progress
        local g = currentColor.G + (nextColor.G - currentColor.G) * progress
        local b = currentColor.B + (nextColor.B - currentColor.B) * progress
        
        stroke.Color = Color3.new(r, g, b)
        
        progress = progress + 0.02 -- Smooth transition speed
        
        if progress >= 1 then
            progress = 0
            currentIndex = nextIndex
            nextIndex = currentIndex % #colors + 1
        end
        
        task.wait(0.016) -- ~60 FPS
    end
end

-- ==================== CLEANUP HỆ THỐNG CŨ ====================
-- Cơ chế Cleanup mới giúp ngắt kết nối dứt điểm khi Execute lần 2
if _G.NuclearBobo5_Cleanup then
    pcall(_G.NuclearBobo5_Cleanup)
end

-- Variables
local character, humanoid, rootPart
local invisible = false
local parts = {}
local connections = {}
local uiVisible = true
local gui, mainFrame

_G.NuclearBobo5_Cleanup = function()
    for _, conn in pairs(connections) do
        pcall(function() conn:Disconnect() end)
    end
    if gui then
        pcall(function() gui:Destroy() end)
    end
end

-- Simple Configuration
local Config = {
    ToggleKey = Enum.KeyCode.X,
    UIToggleKey = Enum.KeyCode.M,
    TeleportHeight = -200000,
    Transparency = 0.7,
    
    -- Orange/Black Theme
    PrimaryColor = Color3.new(1, 0.5, 0),
    SecondaryColor = Color3.new(0.2, 0.2, 0.2),
    BackgroundColor = Color3.fromRGB(10, 10, 10),
    TextColor = Color3.new(1, 1, 1),
    ActiveColor = Color3.new(0, 1, 0),
}

-- Character Setup
local function setupCharacter()
    character = player.Character or player.CharacterAdded:Wait()
    humanoid = character:WaitForChild("Humanoid")
    rootPart = character:WaitForChild("HumanoidRootPart")
    parts = {}
    
    for _, obj in pairs(character:GetDescendants()) do
        if obj:IsA("BasePart") then
            table.insert(parts, obj)
        end
    end
    
    if invisible then
        for _, part in pairs(parts) do
            part.Transparency = Config.Transparency
        end
    end
end

-- Update UI State
local function updateUI()
    if mainFrame then
        local toggleBtn = mainFrame:FindFirstChild("ToggleBtn")
        local statusLabel = mainFrame:FindFirstChild("StatusFrame") and mainFrame.StatusFrame:FindFirstChild("StatusLabel")
        
        if toggleBtn then
            toggleBtn.Text = invisible and "✓" or "X"
            toggleBtn.BackgroundColor3 = invisible and Config.ActiveColor or Config.PrimaryColor
        end
        
        if statusLabel then
            statusLabel.Text = invisible and "Tàng Hình BẬT" or "Tàng Hình TẮT"
            statusLabel.TextColor3 = invisible and Config.ActiveColor or Config.PrimaryColor
        end
    end
end

-- Invisibility Toggle
local function toggleInvisibility()
    invisible = not invisible
    
    for _, part in pairs(parts) do
        if part:IsA("BasePart") then
            part.Transparency = invisible and Config.Transparency or 0
        end
    end
    
    updateUI()
end

-- UI Toggle
local function toggleUI()
    uiVisible = not uiVisible
    if gui then
        gui.Enabled = uiVisible
    end
end

-- Create Ultra Compact Main UI
local function createMainUI()
    if gui then gui:Destroy() end
    
    gui = Instance.new("ScreenGui")
    gui.Name = "NuclearBobo5UI"
    gui.ResetOnSpawn = false
    gui.Parent = player.PlayerGui
    
    -- Ultra Compact Main Frame
    mainFrame = Instance.new("Frame")
    mainFrame.Size = UDim2.new(0, 160, 0, 70)
    mainFrame.Position = UDim2.new(0.5, -80, 0.1, 0)
    mainFrame.BackgroundColor3 = Config.BackgroundColor
    mainFrame.BackgroundTransparency = 0.1
    mainFrame.BorderSizePixel = 0
    mainFrame.Active = true
    mainFrame.Draggable = true
    
    -- Smooth Rainbow Border
    local mainBorder = Instance.new("UIStroke")
    mainBorder.Color = Config.PrimaryColor
    mainBorder.Thickness = 2
    mainBorder.Transparency = 0
    mainBorder.Parent = mainFrame
    
    task.spawn(function()
        createSmoothRainbow(mainBorder)
    end)
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = mainFrame
    
    -- Status Display
    local statusFrame = Instance.new("Frame")
    statusFrame.Name = "StatusFrame"
    statusFrame.Size = UDim2.new(1, -16, 0, 20)
    statusFrame.Position = UDim2.new(0, 8, 0, 8)
    statusFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
    statusFrame.BackgroundTransparency = 0.3
    statusFrame.BorderSizePixel = 0
    
    local statusCorner = Instance.new("UICorner")
    statusCorner.CornerRadius = UDim.new(0, 4)
    statusCorner.Parent = statusFrame
    
    local statusLabel = Instance.new("TextLabel")
    statusLabel.Name = "StatusLabel"
    statusLabel.Size = UDim2.new(1, 0, 1, 0)
    statusLabel.BackgroundTransparency = 1
    statusLabel.Text = "Tàng Hình TẮT"
    statusLabel.TextColor3 = Config.PrimaryColor
    statusLabel.Font = Enum.Font.Gotham
    statusLabel.TextSize = 10
    statusLabel.TextXAlignment = Enum.TextXAlignment.Center
    statusLabel.Parent = statusFrame
    
    statusFrame.Parent = mainFrame

    -- Main Toggle Button
    local toggleBtn = Instance.new("TextButton")
    toggleBtn.Name = "ToggleBtn"
    toggleBtn.Size = UDim2.new(1, -16, 0, 30)
    toggleBtn.Position = UDim2.new(0, 8, 0, 32)
    toggleBtn.Text = "X"
    toggleBtn.BackgroundColor3 = Config.PrimaryColor
    toggleBtn.TextColor3 = Config.TextColor
    toggleBtn.Font = Enum.Font.GothamBold
    toggleBtn.TextSize = 12
    toggleBtn.AutoButtonColor = false
    
    -- Button rainbow stroke
    local toggleStroke = Instance.new("UIStroke")
    toggleStroke.Color = Color3.new(1, 1, 1)
    toggleStroke.Thickness = 1
    toggleStroke.Transparency = 0.7
    toggleStroke.Parent = toggleBtn
    
    task.spawn(function()
        createSmoothRainbow(toggleStroke)
    end)
    
    local btnCorner = Instance.new("UICorner")
    btnCorner.CornerRadius = UDim.new(0, 6)
    btnCorner.Parent = toggleBtn
    
    toggleBtn.MouseButton1Click:Connect(function()
        toggleInvisibility()
    end)
    toggleBtn.Parent = mainFrame
    
    mainFrame.Parent = gui
end

-- Teleport Loop
local function startTeleportLoop()
    connections.Heartbeat = RunService.Heartbeat:Connect(function()
        if invisible and character and rootPart and humanoid then
            local cf = rootPart.CFrame
            local camOffset = humanoid.CameraOffset
            local hidden = cf * CFrame.new(0, Config.TeleportHeight, 0)
            
            rootPart.CFrame = hidden
            humanoid.CameraOffset = hidden:ToObjectSpace(CFrame.new(cf.Position)).Position
            
            RunService.RenderStepped:Wait()
            
            rootPart.CFrame = cf
            humanoid.CameraOffset = camOffset
        end
    end)
end

-- Initialize
setupCharacter()
createMainUI()
startTeleportLoop()

-- Keybinds
connections.KeyToggle = UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Config.ToggleKey then
        toggleInvisibility()
    elseif input.KeyCode == Config.UIToggleKey then
        toggleUI()
    end
end)

-- Character Handling (ĐÃ SỬA LỖI 2 MAIN GUI)
connections.CharacterAdded = player.CharacterAdded:Connect(function()
    invisible = false
    task.wait(0.5) -- Đợi một chút để tải character
    setupCharacter()
    
    -- Chỉ cập nhật lại UI chứ KHÔNG tạo thêm Main GUI mới
    updateUI()
end)

-- Cleanup khi rời game
connections.PlayerLeaving = player:GetPropertyChangedSignal("Parent"):Connect(function()
    if not player.Parent then
        if _G.NuclearBobo5_Cleanup then
            pcall(_G.NuclearBobo5_Cleanup)
        end
    end
end)
