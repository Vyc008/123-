-- ================= HÀM HỖ TRỢ UI =================
local function applyUICorner(instance, radius)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius or 6)
    corner.Parent = instance
end

local function applyUIStroke(instance, color)
    local stroke = Instance.new("UIStroke")
    stroke.Color = color or Color3.fromRGB(70, 70, 75)
    stroke.Thickness = 1.5
    stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    stroke.Parent = instance
end

local function makeDraggable(dragObject, moveObject)
    local dragging, dragInput, dragStart, startPos
    dragObject.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true; dragStart = input.Position; startPos = moveObject.Position
            input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
        end
    end)
    dragObject.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            moveObject.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end

-- ================= LOGIC AUTO FARM =================
-- Hàm thực thi nhặt đồ (Chạy ngầm trong 1 luồng riêng)
local function startFarming()
    task.spawn(function()
        while isFarming do
            local character = player.Character
            local coinsFolder = workspace:FindFirstChild("Coins")
            
            if character and character:FindFirstChild("HumanoidRootPart") and coinsFolder then
                local hrp = character.HumanoidRootPart
                
                for _, coinModel in pairs(coinsFolder:GetChildren()) do
                    -- Nếu đang nhặt mà bị tắt nút, lập tức ngừng vòng lặp này
                    if not isFarming then break end 
                    
                    local coinPart = coinModel:FindFirstChild("Coin")
                    if coinPart and coinPart:IsA("BasePart") then
                        hrp.CFrame = coinPart.CFrame
                        -- Chờ theo thời gian đã thiết lập trên UI
                        task.wait(tpDelay)
                    end
                end
            end
            -- Đợi 0.1s trước khi quét lại toàn bộ map để tránh giật lag máy
            task.wait(0.1)
        end
    end)
end

-- ================= KẾT NỐI SỰ KIỆN UI =================
ToggleBtn.Activated:Connect(function()
    isFarming = not isFarming
    if isFarming then
        ToggleBtn.Text = "AUTO: ON"
        ToggleBtn.BackgroundColor3 = Color3.fromRGB(60, 160, 80)
        startFarming()
    else
        ToggleBtn.Text = "AUTO: OFF"
        ToggleBtn.BackgroundColor3 = Color3.fromRGB(180, 50, 50)
    end
end)

-- Khi người dùng nhập số mới vào ô thời gian
DelayInput.FocusLost:Connect(function()
    local inputNumber = tonumber(DelayInput.Text)
    -- Kiểm tra xem người dùng có nhập đúng số không (chống nhập chữ)
    if inputNumber and inputNumber >= 0 then
        tpDelay = inputNumber
    else
        -- Nếu nhập sai, trả về số cũ
        DelayInput.Text = tostring(tpDelay)
    end
end)
