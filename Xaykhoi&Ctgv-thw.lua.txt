local Lighting = game:GetService("Lighting")

local function clearFog()
	Lighting.FogStart = 0
	Lighting.FogEnd = 1e9
end

local function neuterAtmosphere(a)
	if a and a:IsA("Atmosphere") then
		a.Density = 0
		pcall(function() a.Haze = 0 end)
		pcall(function() a.Glare = 0 end)
		a:GetPropertyChangedSignal("Density"):Connect(function() a.Density = 0 end)
		pcall(function() a:GetPropertyChangedSignal("Haze"):Connect(function() a.Haze = 0 end) end)
		pcall(function() a:GetPropertyChangedSignal("Glare"):Connect(function() a.Glare = 0 end) end)
	end
end

clearFog()
for _, child in ipairs(Lighting:GetChildren()) do
	neuterAtmosphere(child)
end

Lighting.ChildAdded:Connect(neuterAtmosphere)
Lighting:GetPropertyChangedSignal("FogStart"):Connect(clearFog)
Lighting:GetPropertyChangedSignal("FogEnd"):Connect(clearFog)

local function applyStyle(instance, radius)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius or 8)
    corner.Parent = instance
    
    if instance:IsA("GuiButton") or instance:IsA("Frame") then
        local stroke = Instance.new("UIStroke")
        stroke.Color = Color3.fromRGB(60, 60, 65)
        stroke.Thickness = 1.5
        stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
        stroke.Parent = instance
    end
end

-- LOGIC HỆ THỐNG VÀ LỊCH SỬ UNDO (VÔ HẠN LẦN)
-- ==========================================
local targets = {}
local originalStates = {} 
local activeHandles = {} 
local undoStack = {} 
local currentHandleStyle = Enum.HandlesStyle.Movement
local lastCheck = 0
local isClickModeEnabled = false

local function updateUndoButton()
    btnRestore.Text = #undoStack > 0 and "↩️ HOÀN TÁC ("..#undoStack..")" or "↩️ HOÀN TÁC"
end

local function removeHandle(part)
    if activeHandles[part] then
        activeHandles[part]:Destroy()
        activeHandles[part] = nil
    end
end

local function ensureHighlight(part)
    if not part or not part:IsA("BasePart") then return end
    local highlight = part:FindFirstChild("PersistentHighlight")
    if not highlight then
        highlight = Instance.new("Highlight")
        highlight.Name = "PersistentHighlight"
        highlight.Adornee = part
        highlight.FillColor = Color3.fromRGB(255, 255, 255)
        highlight.FillTransparency = 1
        highlight.OutlineColor = Color3.fromRGB(255, 100, 0)
        highlight.OutlineTransparency = 0
        highlight.Parent = part
    end
end

local function removeHighlight(part)
    local highlight = part and part:FindFirstChild("PersistentHighlight")
    if highlight then highlight:Destroy() end
end

local function createHandle(part)
    if activeHandles[part] then return end
    
    local handle
    local startSize = nil
    local startCFrame = nil

    if toolMode == "Move" or toolMode == "Scale" then
        handle = Instance.new("Handles")
        
        if toolMode == "Move" then
            handle.Style = currentHandleStyle 
            handle.Color3 = Color3.fromRGB(50, 255, 50)
        else
            handle.Style = currentHandleStyle == Enum.HandlesStyle.Movement and Enum.HandlesStyle.Resize or currentHandleStyle
            handle.Color3 = Color3.fromRGB(255, 100, 0)
        end
        
        handle.Adornee = part
        handle.Parent = screenGui

        handle.MouseButton1Down:Connect(function()
            startSize = part.Size
            startCFrame = part.CFrame
        end)

        handle.MouseDrag:Connect(function(face, distance)
            if not startSize or not startCFrame then return end
            
            local normal = Vector3.FromNormalId(face)
            local absNormal = Vector3.new(math.abs(normal.X), math.abs(normal.Y), math.abs(normal.Z))
            
            if toolMode == "Move" then
                part.CFrame = startCFrame * CFrame.new(normal * distance)
            elseif toolMode == "Scale" then
                local currentAxisOrigSize = (startSize * absNormal).Magnitude
                local effectiveDistance = distance
                if currentAxisOrigSize + distance < 0.1 then
                    effectiveDistance = 0.1 - currentAxisOrigSize
                end
                
                part.Size = startSize + absNormal * effectiveDistance
                part.CFrame = startCFrame * CFrame.new(normal * (effectiveDistance / 2))
            end
        end)
        
        handle.MouseButton1Up:Connect(function()
            if startSize and startCFrame then
                if startSize ~= part.Size or startCFrame ~= part.CFrame then
                    table.insert(undoStack, {{
                        action = "Transform", 
                        part = part, 
                        oldSize = startSize, 
                        oldCFrame = startCFrame
                    }})
                    updateUndoButton()
                end
            end
        end)

    elseif toolMode == "Rotate" then
        handle = Instance.new("ArcHandles")
        handle.Color3 = Color3.fromRGB(50, 150, 255)
        handle.Adornee = part
        handle.Parent = screenGui

        handle.MouseButton1Down:Connect(function()
            startCFrame = part.CFrame
        end)

        local axisVectors = {
            [Enum.Axis.X] = Vector3.new(1, 0, 0),
            [Enum.Axis.Y] = Vector3.new(0, 1, 0),
            [Enum.Axis.Z] = Vector3.new(0, 0, 1)
        }

        handle.MouseDrag:Connect(function(axis, relativeAngle)
            if not startCFrame then return end
            part.CFrame = startCFrame * CFrame.fromAxisAngle(axisVectors[axis], relativeAngle)
        end)
        
        handle.MouseButton1Up:Connect(function()
            if startCFrame and startCFrame ~= part.CFrame then
                table.insert(undoStack, {{
                    action = "Transform", 
                    part = part, 
                    oldCFrame = startCFrame, 
                    oldSize = part.Size
                }})
                updateUndoButton()
            end
        end)
    end

    activeHandles[part] = handle
end

local function updateAllHandles()
    for part, _ in pairs(targets) do
        removeHandle(part)
        createHandle(part)
    end
end

local function setToolButtonState(activeBtn)
    local btns = {btnMove, btnScale, btnRotate}
    for _, btn in ipairs(btns) do
        btn.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
        btn.TextColor3 = Color3.fromRGB(200, 200, 200)
    end
    activeBtn.BackgroundColor3 = Color3.fromRGB(60, 120, 180)
    activeBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
end

btnMove.MouseButton1Click:Connect(function() toolMode = "Move"; setToolButtonState(btnMove); updateAllHandles() end)
btnScale.MouseButton1Click:Connect(function() toolMode = "Scale"; setToolButtonState(btnScale); updateAllHandles() end)
btnRotate.MouseButton1Click:Connect(function() toolMode = "Rotate"; setToolButtonState(btnRotate); updateAllHandles() end)

toggleClickButton.MouseButton1Click:Connect(function()
    isClickModeEnabled = not isClickModeEnabled
    if isClickModeEnabled then
        toggleClickButton.Text = "CHẾ ĐỘ CHỌN: ĐANG BẬT"
        toggleClickButton.BackgroundColor3 = Color3.fromRGB(50, 150, 50)
    else
        toggleClickButton.Text = "CHẾ ĐỘ CHỌN: TẮT"
        toggleClickButton.BackgroundColor3 = Color3.fromRGB(80, 80, 85)
    end
end)

toggleSizeButton.MouseButton1Click:Connect(function()
    if currentHandleStyle == Enum.HandlesStyle.Movement then
        currentHandleStyle = Enum.HandlesStyle.Resize
        toggleSizeButton.Text = "KÍCH THƯỚC: KHỐI VUÔNG"
        toggleSizeButton.BackgroundColor3 = Color3.fromRGB(140, 80, 180)
    else
        currentHandleStyle = Enum.HandlesStyle.Movement
        toggleSizeButton.Text = "KÍCH THƯỚC: MŨI TÊN"
        toggleSizeButton.BackgroundColor3 = Color3.fromRGB(60, 120, 180)
    end
    updateAllHandles()
end)

-- LOGIC XÓA KHỐI
btnDelete.MouseButton1Click:Connect(function()
    local batch = {}
    for part, _ in pairs(targets) do
        if part and part.Parent then
            table.insert(batch, {action = "Delete", part = part, parent = part.Parent})
            part.Parent = nil 
            removeHighlight(part)
            removeHandle(part)
        end
    end
    
    if #batch > 0 then
        table.insert(undoStack, batch)
        updateUndoButton()
    end
    table.clear(targets)
end)

-- LOGIC KHÔI PHỤC 
btnRestore.MouseButton1Click:Connect(function()
    if #undoStack > 0 then
        local lastBatch = table.remove(undoStack)
        
        for _, item in ipairs(lastBatch) do
            if item.action == "Delete" and item.part then
                item.part.Parent = item.parent 
            elseif item.action == "Transform" and item.part then
                if item.oldSize then item.part.Size = item.oldSize end
                if item.oldCFrame then item.part.CFrame = item.oldCFrame end
            end
        end
        updateUndoButton()
    end
end)

-- ==========================================
-- XỬ LÝ CHUỘT (CHỌN KHỐI)
-- ==========================================
local lastClickTime = 0
local lastClickedPart = nil

Mouse.Button1Down:Connect(function()
    if not isClickModeEnabled then return end 

    local target = Mouse.Target
    if not target then return end
    if target:IsDescendantOf(LocalPlayer.Character) then return end

    local currentTime = tick()
    if target == lastClickedPart and (currentTime - lastClickTime) <= 0.4 then
        if targets[target] then
            targets[target] = nil
            removeHighlight(target)
            removeHandle(target)
        else
            targets[target] = true
            ensureHighlight(target)
            createHandle(target)
            
            if not originalStates[target] then
                originalStates[target] = {Size = target.Size, CFrame = target.CFrame, Parent = target.Parent}
            end
        end
        lastClickedPart = nil 
    else
        lastClickedPart = target
    end
    lastClickTime = currentTime
end)

resetButton.MouseButton1Click:Connect(function()
    for part, state in pairs(originalStates) do
        if part and part:IsA("BasePart") then
            if not part.Parent then part.Parent = state.Parent end 
            part.Size = state.Size
            part.CFrame = state.CFrame
            removeHighlight(part)
            removeHandle(part)
        end
    end
    table.clear(targets)
    table.clear(originalStates)
    table.clear(activeHandles)
    table.clear(undoStack) 
    updateUndoButton()
end)

toggleGuiBtn.MouseButton1Click:Connect(function()
    frame.Visible = not frame.Visible
    toggleGuiBtn.Text = frame.Visible and "👉👌" or "🥵"
end)

RunService.RenderStepped:Connect(function()
    if tick() - lastCheck >= 2 then
        lastCheck = tick()
        for part, _ in pairs(targets) do
            if part and part.Parent and part:IsA("BasePart") then
                if not part:FindFirstChild("PersistentHighlight") then
                    ensureHighlight(part)
                end
            else
                targets[part] = nil
                removeHandle(part)
            end
        end
    end
end)
